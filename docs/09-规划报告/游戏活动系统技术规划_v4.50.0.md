```javascript
// backend/src/services/gameEventWebSocket.js
/**
 * 游戏活动WebSocket服务
 */
class GameEventWebSocketService {
  constructor(io) {
    this.io = io
    this.messageQueue = new Map() // 用户ID -> 消息队列
    this.MAX_QUEUE_SIZE = 100
  }

  /**
   * 广播活动状态变更
   */
  broadcastEventStatusChange(eventId, newStatus) {
    const event = this.io.of('/game-events')
    event.emit('event:statusChanged', {
      eventId,
      newStatus,
      timestamp: new Date().toISOString()
    })
  }

  /**
   * 推送用户进度更新
   */
  pushUserProgressUpdate(userId, progressData) {
    const userSocket = this.findUserSocket(userId)
    if (userSocket) {
      userSocket.emit('event:progressUpdated', progressData)
    } else {
      // 加入消息队列，等待用户重连
      this.enqueueMessage(userId, 'event:progressUpdated', progressData)
    }
  }

  /**
   * 用户重连时补发消息
   */
  onUserReconnect(userId, socket) {
    const messages = this.messageQueue.get(userId) || []
    messages.forEach(msg => {
      socket.emit(msg.event, msg.data)
    })
    this.messageQueue.delete(userId)
  }
}

module.exports = GameEventWebSocketService
```

**服务层实现**：

```javascript
// backend/src/services/gameEventTriggerService.js
/**
 * 游戏活动触发器服务
 */
class GameEventTriggerService {
  /**
   * 创建触发器
   */
  async createTrigger(triggerData) {
    // 验证触发器配置
    await this.validateTriggerConfig(triggerData)
    // 保存触发器
  }

  /**
   * 处理行为触发器
   */
  async handleBehaviorTrigger(userId, actionType, actionData) {
    // 查询匹配的触发器
    const triggers = await this.getMatchingTriggers(actionType)
    
    for (const trigger of triggers) {
      await this.executeTrigger(trigger, userId, actionData)
    }
  }

  /**
   * 执行触发器
   */
  async executeTrigger(trigger, userId, contextData) {
    try {
      // 验证条件
      const conditionMet = await this.checkCondition(trigger, contextData)
      
      if (conditionMet) {
        // 执行动作
        await this.executeAction(trigger, userId, contextData)
        
        // 记录成功
        await this.logTriggerExecution(trigger.id, trigger.event_id, userId, 'success')
      }
    } catch (error) {
      // 记录失败
      await this.logTriggerExecution(trigger.id, trigger.event_id, userId, 'failed', error)
      
      // 重试机制
      if (trigger.retry_count < 3) {
        await this.scheduleRetry(trigger, userId, contextData)
      }
    }
  }

  /**
   * 条件检查
   */
  async checkCondition(trigger, contextData) {
    const condition = trigger.trigger_condition
    
    switch (trigger.trigger_type) {
      case 'threshold_based':
        return this.checkThresholdCondition(condition, contextData)
      case 'behavior_based':
        return this.checkBehaviorCondition(condition, contextData)
      case 'time_based':
        return this.checkTimeCondition(condition)
      default:
        return false
    }
  }
}

module.exports = new GameEventTriggerService()
```

```javascript
/**
 * 游戏活动统计服务
 */
class GameEventStatsService {
  /**
   * 计算活动统计
   */
  async computeEventStats(eventId, date) {
    const stats = {
      event_id: eventId,
      stat_date: date
    }
    
    // 计算参与指标
    stats.unique_participants = await this.getUniqueParticipants(eventId, date)
    stats.total_participants = await this.getTotalParticipants(eventId, date)
    stats.new_participants = await this.getNewParticipants(eventId, date)
    
    // 计算完成指标
    stats.tasks_completed = await this.getTasksCompleted(eventId, date)
    stats.completion_rate = await this.calculateCompletionRate(eventId, date)
    
    // 计算转化指标
    stats.reward_claimed_count = await this.getRewardClaimed(eventId, date)
    stats.conversion_rate = await this.calculateConversionRate(eventId, date)
    
    // 计算留存指标
    stats.day1_retention = await this.calculateRetention(eventId, date, 1)
    stats.day7_retention = await this.calculateRetention(eventId, date, 7)
    
    return stats
  }

  /**
   * 计算漏斗数据
   */
  async computeEventFunnel(eventId, date) {
    const steps = [
      'saw_event',
      'clicked_event',
      'joined_event',
      'completed_task',
      'claimed_reward'
    ]
    
    const funnelData = []
    
    for (let i = 0; i < steps.length; i++) {
      const userCount = await this.getStepUserCount(eventId, date, steps[i])
      const prevCount = i > 0 ? funnelData[i-1].user_count : userCount
      const conversionRate = prevCount > 0 ? (userCount / prevCount * 100).toFixed(2) : 0
      
      funnelData.push({
        event_id: eventId,
        funnel_step: steps[i],
        user_count: userCount,
        conversion_rate: conversionRate,
        stat_date: date
      })
    }
    
    return funnelData
  }
}

module.exports = new GameEventStatsService()
```

```javascript
/**
 * 游戏活动模板服务
 */
class GameEventTemplateService {
  /**
   * 从模板创建活动
   */
  async createEventFromTemplate(templateId, templateVariables) {
    const template = await this.getTemplate(templateId)
    
    // 验证变量
    await this.validateVariables(template, templateVariables)
    
    // 渲染模板配置
    const renderedConfig = await this.renderTemplate(template, templateVariables)
    
    // 创建活动
    const eventData = {
      event_name: renderedConfig.event_name,
      event_description: renderedConfig.event_description,
      start_time: renderedConfig.start_time,
      end_time: renderedConfig.end_time,
      event_config: renderedConfig.event_config,
      is_active: true
    }
    
    return await gameEventService.createEvent(eventData)
  }

  /**
   * 模板渲染
   */
  async renderTemplate(template, variables) {
    const config = JSON.parse(JSON.stringify(template.template_config))
    
    // 变量替换
    config = this.replaceVariables(config, variables)
    
    // 日期计算
    config = this.processDateExpressions(config, variables)
    
    return config
  }

  /**
   * 替换变量
   */
  replaceVariables(obj, variables) {
    if (typeof obj === 'string') {
      return obj.replace(/\{\{(\w+)\}\}/g, (match, varName) => {
        return variables[varName] || match
      })
    } else if (typeof obj === 'object' && obj !== null) {
      for (const key in obj) {
        obj[key] = this.replaceVariables(obj[key], variables)
      }
    }
    return obj
  }
}
```

##### 阶段一：任务调度框架（2周）

**技术选型**：`node-schedule` + Redis（分布式锁）

```javascript
// backend/src/services/gameEventSchedulerService.js
const schedule = require('node-schedule')
const Redis = require('ioredis')

/**
 * 游戏活动定时任务调度服务
 */
class GameEventSchedulerService {
  constructor() {
    this.redis = new Redis(process.env.REDIS_URL || 'redis://localhost:6379')
    this.jobs = new Map()
  }

  /**
   * 调度活动开始任务
   */
  async scheduleEventStart(eventId, startTime) {
    const jobId = `event_start:${eventId}`
    
    // 使用Redis分布式锁
    const lockKey = `lock:${jobId}`
    const lockAcquired = await this.redis.set(lockKey, '1', 'NX', 'EX', 60)
    
    if (lockAcquired) {
      const job = schedule.scheduleJob(jobId, startTime, async () => {
        await this.executeEventStart(eventId)
      })
      
      this.jobs.set(jobId, job)
    }
  }

  /**
   * 执行活动开始
   */
  async executeEventStart(eventId) {
    try {
      console.log(`活动 ${eventId} 开始执行`)
      
      // 更新活动状态
      await gameEventService.updateEventStatus(eventId, 'active')
      
      // 发送通知
      await this.sendEventStartNotifications(eventId)
      
      // 记录成功
      await this.logTaskExecution('event_start', eventId, 'success')
    } catch (error) {
      console.error(`活动 ${eventId} 开始失败`, error)
      
      // 记录失败
      await this.logTaskExecution('event_start', eventId, 'failed', error.message)
      
      // 重试策略
      await this.scheduleRetry('event_start', eventId, 1)
    }
  }

  /**
   * 定时任务列表
   */
  getAvailableTasks() {
    return [
      { id: 'event_start', name: '活动开始' },
      { id: 'event_end', name: '活动结束' },
      { id: 'daily_reset', name: '日常任务重置' },
      { id: 'weekly_reset', name: '周常任务重置' },
      { id: 'stats_compute', name: '统计数据计算' }
    ]
  }
}

module.exports = new GameEventSchedulerService()
```
